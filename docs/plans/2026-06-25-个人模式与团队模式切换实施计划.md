# 个人模式与团队模式切换实施计划

## 概述

### 1. 总体目标和范围

为 shared view 增加项目级的 `个人模式 / 团队模式` 切换，但模式值只保存在当前用户的命名 profile 文件中，不写入团队共享配置。`团队模式` 保持现有 draft + 显式发布语义；`个人模式` 下 shared view 的常规编辑直接保存到 `shared-views.json`，失败时降级为轻量重试入口，不重新恢复常驻“保存团队共享视图”按钮。

### 2. 各阶段任务概要

- 第一阶段：补测试契约，锁定 profile 字段、设置面板入口、模式切换迁移前置条件。
- 第二阶段：扩展 `UserViewProfile` 与 `view-profile.mjs`，把 `sharedViewCollaborationMode` 打通到个人配置读写链路。
- 第三阶段：在 Toolbar 设置浮框中加入协作模式区块，并补对应样式与禁用态说明。
- 第四阶段：实现 `团队模式 -> 个人模式` 切换时自动发布当前 drafts，再持久化模式切换结果。
- 第五阶段：把 shared view 的内容编辑与结构拖拽在个人模式下改为直存，并用可执行 e2e 覆盖。
- 第六阶段：实现个人模式直存失败后的 pending/retry 降级，不再重新暴露团队保存按钮。
- 第七阶段：同步更新文档并跑最小验证集，收尾确认方案闭环。

### 3. 整体结构框架

本次改动分为四层：

- 数据模型层：扩展 `UserViewProfile`，并在 `view-profile.mjs`、`App.tsx` 的 normalize / empty helper 中统一模式值。
- 交互层：在现有设置浮框中新增协作模式选择；无命名 profile 时禁用 `个人模式` 并给出说明。
- 状态层：在 `App.tsx` 中区分团队模式与个人模式的 shared view 保存策略，补切模式迁移、直存与失败重试逻辑。
- 验证层：用 `node:test` 锁定类型和源码契约，用 Playwright 覆盖模式切换、直存、失败降级的关键回归链路。

## 技术栈

React + TypeScript、Node.js、`node:test`、Playwright、现有 `saveViewProfile` / `loadViewProfile` / `saveSharedViews` / `saveSharedViewDraftsToConfig` / `SaveCoordinator`

## 当前执行假设

- `sharedViewCollaborationMode` 只在命名 profile 中长期持久化；无 profile 时不新增 localStorage 真相
- `团队模式 -> 个人模式` 是有副作用的切换：切换前先发布当前 profile 下所有 collection 的 shared drafts，避免全局模式切换后残留旧团队草稿
- 个人模式下只要 shared view 正式配置保存成功，就不再写 `viewDrafts / viewOrderDrafts / structureDrafts`
- 个人模式直存失败时，允许临时存在 pending draft，但该 pending draft 只作为恢复机制，不驱动常驻团队保存按钮显示
- 个人模式 shared view 直存必须经过单一串行保存入口，避免连续筛选/排序/拖拽时用陈旧 `sharedViewsConfig` 快照覆盖较新的编辑
- 个人模式下 `重置` 不再表示“回退未发布团队草稿”；默认禁用，仅在直存失败产生 pending fallback 时作为清理恢复草稿的入口
- 当前已直写的 shared view 显式命令（新建/复制等）在两种模式下都继续直写；本轮重点是把内容和结构编辑补齐到个人模式直存语义

```ts
async function publishCurrentSharedDraftsForModeSwitch() {
  const currentDraftState = currentSharedViewDraftState();
  const collectionKeys = collectDraftCollectionKeys(currentDraftState);
  if (!collectionKeys.length) return;

  let nextConfig = sharedViewsConfig;
  let nextDraftState = currentDraftState;
  for (const collectionKey of collectionKeys) {
    const activeViewId = nextDraftState.lastActiveViews?.[collectionKey]
      ?? (collectionKey === activeCollectionKey ? activeSharedView?.id ?? null : null);
    if (!activeViewId) continue;
    nextConfig = saveSharedViewDraftsToConfig(nextConfig, nextDraftState, collectionKey, activeViewId);
    nextDraftState = clearCollectionSharedDrafts(nextDraftState, collectionKey);
  }

  await saveSharedViews(nextConfig, activeProjectId);
  setSharedViewsConfig(nextConfig);
  updateSharedViewDraftState(nextDraftState);
}
```

```ts
function collectDraftCollectionKeys(draftState: SharedViewDraftState): string[] {
  return [...new Set([
    ...Object.keys(draftState.viewDrafts ?? {}),
    ...Object.keys(draftState.viewOrderDrafts ?? {}),
    ...Object.keys(draftState.structureDrafts ?? {}),
  ])];
}

function clearCollectionSharedDrafts(draftState: SharedViewDraftState, collectionKey: string): SharedViewDraftState {
  const nextLastActiveViews = { ...draftState.lastActiveViews };
  const nextViewDrafts = { ...draftState.viewDrafts };
  const nextViewOrderDrafts = { ...draftState.viewOrderDrafts };
  const nextStructureDrafts = { ...(draftState.structureDrafts ?? {}) };
  delete nextViewDrafts[collectionKey];
  delete nextViewOrderDrafts[collectionKey];
  delete nextStructureDrafts[collectionKey];
  return {
    lastActiveViews: nextLastActiveViews,
    viewDrafts: nextViewDrafts,
    viewOrderDrafts: nextViewOrderDrafts,
    structureDrafts: nextStructureDrafts,
  };
}
```

```ts
async function handleChangeSharedViewCollaborationMode(mode: "team" | "personal") {
  if (!selectedViewProfileName) return;
  const currentMode = resolveSharedViewCollaborationMode(selectedViewProfileName, selectedViewProfileRef.current);
  if (mode === currentMode) return;

  setCommandSaving(true);
  setStatus("");
  try {
    if (mode === "personal") {
      await publishCurrentSharedDraftsForModeSwitch();
    }
    const nextProfile = mutateSelectedViewProfile((draft) => {
      draft.sharedViewCollaborationMode = mode;
    });
    await commitProfileSave(selectedViewProfileName, nextProfile);
    setStatus(mode === "personal" ? "已切换到个人模式" : "已切换到团队模式");
  } catch (error) {
    setStatus(mode === "personal"
      ? "切换到个人模式前未能发布当前共享视图草稿"
      : (error instanceof Error ? error.message : String(error)));
  } finally {
    setCommandSaving(false);
  }
}
```

```ts
async function enqueueSharedViewDirectSave(
  mutator: (config: SharedViewsConfig) => SharedViewsConfig,
  draftFallback: SharedViewDraftState,
) {
  const baseConfig = sharedViewsConfigRef.current;
  const nextConfig = mutator(baseConfig);
  try {
    await saveSharedViews(nextConfig, activeProjectId);
    sharedViewsConfigRef.current = nextConfig;
    setSharedViewsConfig(nextConfig);
    setSharedViewDirectSavePending(false);
    sharedViewDirectSaveRetryRef.current = null;
  } catch (error) {
    setSharedViewDirectSavePending(true);
    updateSharedViewDraftState(draftFallback);
    sharedViewDirectSaveRetryRef.current = async () => {
      await saveSharedViews(nextConfig, activeProjectId);
      sharedViewsConfigRef.current = nextConfig;
      setSharedViewsConfig(nextConfig);
      updateSharedViewDraftState(clearCollectionSharedDrafts(draftFallback, activeCollectionKey!));
      setSharedViewDirectSavePending(false);
      sharedViewDirectSaveRetryRef.current = null;
    };
    setStatus("共享视图自动保存失败");
  }
}
```
