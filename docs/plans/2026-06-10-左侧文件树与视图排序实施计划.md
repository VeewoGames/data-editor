# 左侧文件树与视图排序 Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use `superpowers:subagent-driven-development` (recommended) or `superpowers:executing-plans` to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** 让左侧 `Files` 从平铺文件列表升级为真实文件树导航，支持同父节点内文件夹/文件混排拖拽排序、展开状态持久化，以及基于树顺序的默认打开文件规则。

**Architecture:** 服务端继续只返回扁平 `DataFile[]`。前端新增树构建与树偏好纯函数，把 `DataFile[]` 投影成 source/folder/file 树，并用 `childOrderByParent + expandedNodeIds` 控制显示顺序与展开状态。`App.tsx` 负责持久化、默认打开文件选择，并彻底停止用旧 `fileOrder` 驱动 Sidebar；`Sidebar.tsx` 只负责树渲染和交互。

**Tech Stack:** React 18, TypeScript, Vite, Node.js `node:test`, Playwright, browser `localStorage`, project view profile JSON.

```js
function normalizePath(value) {
  return String(value ?? "").replaceAll("\\", "/").replace(/^\/+|\/+$/g, "");
}

function normalizeStringArray(value) {
  if (!Array.isArray(value)) return [];
  const seen = new Set();
  const result = [];
  for (const item of value) {
    const normalized = String(item ?? "").trim();
    if (!normalized || seen.has(normalized)) continue;
    seen.add(normalized);
    result.push(normalized);
  }
  return result;
}

function compareNodes(left, right) {
  return `${left.kind}:${left.label}`.localeCompare(`${right.kind}:${right.label}`, undefined, { numeric: true });
}

export function buildSidebarTreePreferences(value = {}) {
  const childOrderByParent = {};
  for (const [parentId, order] of Object.entries(value.childOrderByParent ?? {})) {
    const normalizedParentId = String(parentId ?? "").trim();
    if (!normalizedParentId) continue;
    childOrderByParent[normalizedParentId] = normalizeStringArray(order);
  }
  return {
    childOrderByParent,
    expandedNodeIds: normalizeStringArray(value.expandedNodeIds),
  };
}

export function buildSidebarTree(files) {
  const sourceMap = new Map();
  for (const file of files) {
    const dataSourceId = String(file.dataSourceId ?? "").trim();
    const dataSourceLabel = String(file.dataSourceLabel ?? dataSourceId).trim();
    const sourceId = `source:${dataSourceId}`;
    let source = sourceMap.get(sourceId);
    if (!source) {
      source = { kind: "source", id: sourceId, label: dataSourceLabel, parentId: null, children: [], _childMap: new Map() };
      sourceMap.set(sourceId, source);
    }
    const displayPath = normalizePath(file.displayPath ?? file.path);
    const parts = displayPath.split("/").filter(Boolean);
    let parent = source;
    let folderPath = "";
    for (const part of parts.slice(0, -1)) {
      folderPath = folderPath ? `${folderPath}/${part}` : part;
      const folderId = `folder:${dataSourceId}/${folderPath}`;
      let folder = parent._childMap.get(folderId);
      if (!folder) {
        folder = {
          kind: "folder",
          id: folderId,
          label: part,
          parentId: parent.id,
          dataSourceId,
          folderPath,
          children: [],
          _childMap: new Map(),
        };
        parent._childMap.set(folderId, folder);
        parent.children.push(folder);
      }
      parent = folder;
    }
    parent.children.push({
      kind: "file",
      id: `file:${normalizePath(file.path)}`,
      label: parts.at(-1) ?? normalizePath(file.path),
      parentId: parent.id,
      dataSourceId,
      filePath: normalizePath(file.path),
      file,
    });
  }
  return [...sourceMap.values()].map(finalizeNode).sort(compareNodes);
}

function finalizeNode(node) {
  return {
    ...node,
    children: (node.children ?? []).map(finalizeNode).sort(compareNodes),
  };
}

export function applySidebarTreePreferences(tree, preferences) {
  const normalized = buildSidebarTreePreferences(preferences);
  return tree.map((node) => applyNodePreferences(node, normalized.childOrderByParent));
}

function applyNodePreferences(node, childOrderByParent) {
  if (!node.children?.length) return node;
  const childMap = new Map(node.children.map((child) => [child.id, child]));
  const orderedIds = childOrderByParent[node.id] ?? [];
  const orderedChildren = [];
  const seen = new Set();
  for (const childId of orderedIds) {
    const child = childMap.get(childId);
    if (!child || seen.has(childId)) continue;
    seen.add(childId);
    orderedChildren.push(child);
  }
  for (const child of node.children) {
    if (seen.has(child.id)) continue;
    seen.add(child.id);
    orderedChildren.push(child);
  }
  return {
    ...node,
    children: orderedChildren.map((child) => applyNodePreferences(child, childOrderByParent)),
  };
}

export function reorderSidebarSiblingIds(siblingIds, sourceId, targetId, placement = "before") {
  const base = normalizeStringArray(siblingIds);
  const sourceIndex = base.indexOf(sourceId);
  const targetIndex = base.indexOf(targetId);
  if (sourceIndex < 0 || targetIndex < 0 || sourceId === targetId) return base;
  const next = base.filter((id) => id !== sourceId);
  const insertIndex = placement === "after"
    ? Math.min(targetIndex + 1, next.length)
    : Math.max(targetIndex - (sourceIndex < targetIndex ? 1 : 0), 0);
  next.splice(insertIndex, 0, sourceId);
  return next;
}

export function findSidebarNodeAncestorIds(tree, filePath) {
  const targetId = `file:${normalizePath(filePath)}`;
  return findAncestors(tree, targetId, []) ?? [];
}

function findAncestors(nodes, targetId, trail) {
  for (const node of nodes) {
    if (node.id === targetId) return trail;
    const nextTrail = node.kind === "file" ? trail : [...trail, node.id];
    const nested = findAncestors(node.children ?? [], targetId, nextTrail);
    if (nested) return nested;
  }
  return null;
}

export function findSidebarFallbackFilePath(tree, preferredPath) {
  const normalizedPreferredPath = normalizePath(preferredPath);
  const preferredId = normalizedPreferredPath ? `file:${normalizedPreferredPath}` : null;
  const orderedFiles = [];
  collectFiles(tree, orderedFiles);
  if (preferredId) {
    const preferredNode = orderedFiles.find((node) => node.id === preferredId);
    if (preferredNode) return preferredNode.filePath;
  }
  return orderedFiles[0]?.filePath ?? null;
}

function collectFiles(nodes, result) {
  for (const node of nodes) {
    if (node.kind === "file") {
      result.push(node);
      continue;
    }
    collectFiles(node.children ?? [], result);
  }
}

const sidebarTreePrefsStorageKey = "data-editor:__sidebar-tree-prefs";

export function emptySidebarTreePreferences() {
  return { childOrderByParent: {}, expandedNodeIds: [] };
}

export function normalizeSidebarTreePreferences(value) {
  const childOrderByParent = {};
  for (const [parentId, order] of Object.entries(value?.childOrderByParent ?? {})) {
    const normalizedParentId = String(parentId ?? "").trim();
    if (!normalizedParentId) continue;
    childOrderByParent[normalizedParentId] = normalizeStringArray(order);
  }
  return {
    childOrderByParent,
    expandedNodeIds: normalizeStringArray(value?.expandedNodeIds),
  };
}

export function readLocalSidebarTreePreferences(localStorage) {
  const rawValue = localStorage.getItem(sidebarTreePrefsStorageKey);
  if (!rawValue) return emptySidebarTreePreferences();
  try {
    return normalizeSidebarTreePreferences(JSON.parse(rawValue));
  } catch {
    return emptySidebarTreePreferences();
  }
}

export function writeLocalSidebarTreePreferences(localStorage, value) {
  const normalized = normalizeSidebarTreePreferences(value);
  if (Object.keys(normalized.childOrderByParent).length === 0 && normalized.expandedNodeIds.length === 0) {
    localStorage.removeItem(sidebarTreePrefsStorageKey);
    return;
  }
  localStorage.setItem(sidebarTreePrefsStorageKey, JSON.stringify(normalized));
}
```

- [ ] **Step 4: 新建树拖拽 hook，显式定义 preview 算法，不留未定义函数**

Create `src/components/useSidebarTreeDragReorder.ts`:

```ts
import { useRef, useState, type PointerEvent as ReactPointerEvent } from "react";
import { createOneDimensionalDragSession } from "../drag/one-dimensional-dnd.mjs";
import { reorderSidebarSiblingIds } from "../sidebar-tree.mjs";

function projectSiblingPreview(itemRefs: Record<string, HTMLElement | null>, siblingIds: string[], sourceId: string, clientY: number) {
  let targetId = null;
  let placement: "before" | "after" = "before";
  for (const siblingId of siblingIds) {
    if (siblingId === sourceId) continue;
    const element = itemRefs[siblingId];
    if (!element) continue;
    const rect = element.getBoundingClientRect();
    if (clientY >= rect.top && clientY <= rect.bottom) {
      targetId = siblingId;
      placement = clientY < rect.top + rect.height / 2 ? "before" : "after";
      break;
    }
  }
  if (!targetId) return siblingIds;
  return reorderSidebarSiblingIds(siblingIds, sourceId, targetId, placement);
}

export function useSidebarTreeDragReorder({ itemRefs, onCommitOrder }) {
  const [draggingNodeId, setDraggingNodeId] = useState<string | null>(null);
  const [previewSiblingIds, setPreviewSiblingIds] = useState<string[] | null>(null);
  const suppressNextClickRef = useRef(false);
  const dragSessionRef = useRef<null | {
    pointerId: number;
    siblingIds: string[];
    session: ReturnType<typeof createOneDimensionalDragSession>;
  }>(null);

  function beginDrag(event: ReactPointerEvent<HTMLElement>, nodeId: string, siblingIds: string[]) {
    event.currentTarget.setPointerCapture(event.pointerId);
    dragSessionRef.current = {
      pointerId: event.pointerId,
      siblingIds,
      session: createOneDimensionalDragSession({
        pointerId: event.pointerId,
        startX: event.clientX,
        startY: event.clientY,
        onStart: () => {
          suppressNextClickRef.current = true;
          setDraggingNodeId(nodeId);
        },
        onPreview: ({ clientY }) => {
          const nextOrder = projectSiblingPreview(itemRefs.current, siblingIds, nodeId, clientY);
          setPreviewSiblingIds(nextOrder);
          return nextOrder;
        },
        onCommit: (nextOrder) => {
          if (JSON.stringify(nextOrder) !== JSON.stringify(siblingIds)) onCommitOrder(nextOrder);
        },
        onCancel: () => {
          setDraggingNodeId(null);
          setPreviewSiblingIds(null);
        },
      }),
    };
  }

  function moveDrag(event: ReactPointerEvent<HTMLElement>) {
    const current = dragSessionRef.current;
    if (!current || current.pointerId !== event.pointerId) return;
    const result = current.session.move({ clientX: event.clientX, clientY: event.clientY });
    if (result.started) event.preventDefault();
  }

  function endDrag(event: ReactPointerEvent<HTMLElement>) {
    const current = dragSessionRef.current;
    if (!current || current.pointerId !== event.pointerId) return;
    const started = current.session.started;
    current.session.release();
    dragSessionRef.current = null;
    setDraggingNodeId(null);
    setPreviewSiblingIds(null);
    if (started) {
      window.setTimeout(() => {
        suppressNextClickRef.current = false;
      }, 0);
    } else {
      suppressNextClickRef.current = false;
    }
  }

  function cancelDrag(event: ReactPointerEvent<HTMLElement>) {
    const current = dragSessionRef.current;
    if (!current || current.pointerId !== event.pointerId) return;
    current.session.cancel();
    dragSessionRef.current = null;
    setDraggingNodeId(null);
    setPreviewSiblingIds(null);
    suppressNextClickRef.current = false;
  }

  return { beginDrag, moveDrag, endDrag, cancelDrag, draggingNodeId, previewSiblingIds, suppressNextClickRef };
}
```
